/* Agregar en pantalla.c y declarar en pantalla.h */
void pantalla_put_at(int x, int y, char c, unsigned char color)
{
    volatile char *video = (volatile char *)0xB8000;
    int pos;

    if (x < 0 || x >= 80) return;
    if (y < 0 || y >= 25) return;

    pos = y * 80 + x;
    video[pos * 2] = c;
    video[pos * 2 + 1] = color;
}
